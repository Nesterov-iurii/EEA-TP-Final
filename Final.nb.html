Data loading and cleaning

Download dataset, convert to proper format. Dataset link: https://zenodo.org/records/4656049

# Replace 'your_file_path.tsf' with the actual path to your TSF file
file_path <- 'data/us_births_dataset.tsf'
# Read the lines starting from the line after '@data'
raw_data <- read.table(file_path, header = TRUE, sep = ',', stringsAsFactors = FALSE)
# Extract the date from the seventh row and replace the row value
first_date_value <- str_match(raw_data[7, 1], "^T\\d+:(\\d{4}-\\d{2}-\\d{2}) \\d{2}-\\d{2}-\\d{2}:\\d+")[, 2]
raw_data[7, 1] <- as.numeric(str_match(raw_data[7, 1], "\\d+$")[, 1])
# Drop the first 6 rows
raw_data <- raw_data[-(1:6), ]
# Create a tibble with the required columns
birth_data <- as_tibble(raw_data, rownames='date')
# Rename the 'value' column to 'num_births'
birth_data <- rename(birth_data, num_births = value)
# Create a 'Date' column starting from the first date
birth_data$date <- seq(as.Date(first_date_value), by = '1 day', length.out = nrow(birth_data))
birth_data <- birth_data %>%
  mutate(num_births = as.integer(num_births))
# Print the modified tibble
print(birth_data)

Measuring results

MAPE Create reusable function to calculate MAPE

evaluate_mape <- function(actual_values, predicted_values) {
  absolute_percentage_errors <- abs((actual_values - predicted_values) / actual_values) * 100
  mape <- mean(absolute_percentage_errors, na.rm = TRUE)
  return(mape)
}
